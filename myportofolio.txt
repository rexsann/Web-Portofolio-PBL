  <script>
    const form = document.getElementById("portfolioForm");
    const uploadSection = document.getElementById("uploadSection");
    const portfolioSection = document.getElementById("portfolioSection");
    const showNama = document.getElementById("showNama");
    const showNim = document.getElementById("showNim");
    const showJudul = document.getElementById("showJudul");
    const showDeskripsi = document.getElementById("showDeskripsi");
    const showKategori = document.getElementById("showKategori");
    const showFile = document.getElementById("showFile");
    const posterContainer = document.getElementById("posterContainer");
    const galleryContainer = document.getElementById("galleryContainer");
    const youtubeFrame = document.getElementById("youtubeFrame");
    const editBtn = document.getElementById("editBtn");
    const deleteBtn = document.getElementById("deleteBtn");
    const commentForm = document.getElementById("commentForm");
    const commentList = document.getElementById("commentList");

    let storedData = {};

    form.addEventListener("submit", (e) => {
      e.preventDefault();

      const nama = document.getElementById("nama").value;
      const nim = document.getElementById("nim").value;
      const judul = document.getElementById("judul").value;
      const deskripsi = document.getElementById("deskripsi").value;
      const kategori = document.getElementById("kategori").value;
      const file = document.getElementById("file").files[0];
      const youtube = document.getElementById("youtube").value;
      const posters = document.getElementById("poster").files;
      const gallery = document.getElementById("gallery").files;

      if (!file) {
        alert("Pilih file utama (PDF) terlebih dahulu!");
        return;
      }

      const fileURL = URL.createObjectURL(file);
      let videoId = "";
      if (youtube.includes("watch?v=")) videoId = youtube.split("watch?v=")[1].split("&")[0];
      else if (youtube.includes("youtu.be/")) videoId = youtube.split("youtu.be/")[1].split("?")[0];

      storedData = { nama, nim, judul, deskripsi, kategori, fileURL, youtube, posters: [], gallery: [] };

      const readImages = (files, targetArray, callback) => {
        let count = 0;
        if (files.length === 0) return callback();
        for (let f of files) {
          const reader = new FileReader();
          reader.onload = (e) => {
            targetArray.push(e.target.result);
            count++;
            if (count === files.length) callback();
          };
          reader.readAsDataURL(f);
        }
      };

      readImages(posters, storedData.posters, () => {
        readImages(gallery, storedData.gallery, () => {
          // Simpan ke kategori yang sesuai
          let key = "";
          if (kategori === "Aplikasi Web") key = "aplikasiWebProjects";
          else if (kategori === "Aplikasi Mobile") key = "aplikasiMobileProjects";
          else if (kategori === "Aplikasi IoT") key = "aplikasiIoTProjects";
          else if (kategori === "Artificial Intelligence") key = "aiProjects";

          if (key) {
            const data = JSON.parse(localStorage.getItem(key)) || [];
            data.push(storedData);
            localStorage.setItem(key, JSON.stringify(data));
          }

          // Simpan ke portofolio saya
          const mine = JSON.parse(localStorage.getItem("myPortfolios")) || [];
          mine.push(storedData);
          localStorage.setItem("myPortfolios", JSON.stringify(mine));

          tampilkanHasil();
          alert("✅ Portofolio berhasil diupload dan masuk ke kategori!");
        });
      });
    });

    function tampilkanHasil() {
      const p = storedData;
      showNama.textContent = p.nama;
      showNim.textContent = p.nim;
      showJudul.textContent = p.judul;
      showDeskripsi.textContent = p.deskripsi;
      showKategori.textContent = p.kategori;
      showFile.href = p.fileURL;
      showFile.textContent = "Lihat Dokumen";

      posterContainer.innerHTML = p.posters.map(src => `<div class='col-md-4'><img src='${src}' class='img-fluid rounded shadow'></div>`).join("");
      galleryContainer.innerHTML = p.gallery.map(src => `<div class='col-md-3 col-6'><img src='${src}' class='img-fluid rounded shadow-sm'></div>`).join("");
      youtubeFrame.innerHTML = p.youtube
        ? `<iframe src="https://www.youtube.com/embed/${p.youtube.split("v=")[1] || p.youtube.split(".be/")[1]}" allowfullscreen></iframe>`
        : `<p class="text-light">Tidak ada video</p>`;

      uploadSection.classList.add("hidden");
      portfolioSection.classList.remove("hidden");
      window.scrollTo(0, 0);
    }

    editBtn.addEventListener("click", () => {
      uploadSection.classList.remove("hidden");
      portfolioSection.classList.add("hidden");
      document.getElementById("nama").value = storedData.nama;
      document.getElementById("nim").value = storedData.nim;
      document.getElementById("judul").value = storedData.judul;
      document.getElementById("deskripsi").value = storedData.deskripsi;
      document.getElementById("kategori").value = storedData.kategori;
      document.getElementById("youtube").value = storedData.youtube;
    });

    deleteBtn.addEventListener("click", () => {
      if (confirm("Apakah kamu yakin ingin menghapus portofolio ini?")) {
        storedData = {};
        form.reset();
        commentList.innerHTML = "";
        portfolioSection.classList.add("hidden");
        uploadSection.classList.remove("hidden");
        alert("✅ Portofolio berhasil dihapus!");
      }
    });

    commentForm.addEventListener("submit", (e) => {
      e.preventDefault();
      const text = document.getElementById("commentText").value.trim();
      if (text) {
        const div = document.createElement("div");
        div.classList.add("comment-box");
        div.innerHTML = `<span>${text}</span>`;
        commentList.prepend(div);
        commentForm.reset();
      }
    });
  </script>
